%================ SLIDE 1 =================
\begin{frame}{Ý tưởng DFS}
\begin{itemize}
    \item Duyệt theo chiều sâu (depth)
    \item Đi càng sâu càng tốt trước
    \item Sử dụng đệ quy hoặc stack
\end{itemize}
\end{frame}

%================ SLIDE 2 =================
\begin{frame}{Các bước thực hiện}
\begin{enumerate}
    \item Bắt đầu từ đỉnh nguồn
    \item Đánh dấu đã thăm
    \item Duyệt các đỉnh kề:
    \begin{itemize}
        \item Nếu chưa thăm → gọi đệ quy
    \end{itemize}
\end{enumerate}
\end{frame}

%================ SLIDE 3 =================
\begin{frame}{Phân tích}
\begin{itemize}
    \item Mỗi đỉnh duyệt 1 lần
    \item Mỗi cạnh xét tối đa 2 lần
    \item Thời gian: $O(n + m)$
    \item Bộ nhớ: $O(n)$
\end{itemize}
\end{frame}

%================ SLIDE 4 =================
\begin{frame}[fragile]{Code DFS C++}
\begin{lstlisting}[language=C++]
#include <bits/stdc++.h>
using namespace std;
vector<int> a[1005];
bool d[1005];
int n,m;

void dfs(int u){
    d[u]=1;
    cout<<u<<" ";
    for(int v:a[u]){
        if(!d[v]){
            dfs(v);
        }
    }
}

int main(){
    cin>>n>>m;
    for(int i=1;i<=m;i++){
        int u,v; cin>>u>>v;
        a[u].push_back(v);
        a[v].push_back(u);
    }
    dfs(1);
}
\end{lstlisting}
\end{frame}
